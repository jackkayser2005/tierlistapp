     1‚Üí     1‚Üí/**
     2‚Üí     2‚Üí * RankForge Room Service  ‚Äî  Real-time collaborative tier-list server.
     3‚Üí     3‚Üí * --------------------------------------------------------------
     4‚Üí     4‚Üí * Standalone socket.io mini-service that powers real-time collaboration
     5‚Üí     5‚Üí * on a shared RankForge tier-list board. Supports up to 10 users per room
     6‚Üí     6‚Üí * with presence, identity, activity log, heartbeats, idempotency (seq),
     7‚Üí     7‚Üí * and reconnection resync.
     8‚Üí     8‚Üí *
     9‚Üí     9‚Üí * Runs on a FIXED port 3003. The Next.js frontend connects via the Caddy
    10‚Üí    10‚Üí * gateway using `io("/?XTransformPort=3003")` ‚Äî the gateway forwards based
    11‚Üí    11‚Üí * on that query param, so we just bind :3003 and accept all origins.
    12‚Üí    12‚Üí *
    13‚Üí    13‚Üí * State is in-memory only (no database). Rooms are dropped on restart.
    14‚Üí    14‚Üí *
    15‚Üí    15‚Üí * Lifecycle:
    16‚Üí    16‚Üí *   1. Client connects (socket established, but no user yet).
    17‚Üí    17‚Üí *   2. Client emits `identity` { name, color }   -> stored on socket.data.
    18‚Üí    18‚Üí *   3. Client emits `room:join` { roomId, board? } -> becomes a full member.
    19‚Üí    19‚Üí *   4. Client emits `heartbeat` every ~20s        -> refreshes lastSeen.
    20‚Üí    20‚Üí *   5. Server sweeps every 15s; any member with lastSeen > 45s is force-
    21‚Üí    21‚Üí *      disconnected (reclaims dead sessions).
    22‚Üí    22‚Üí *   6. On disconnect: member removed, presence re-broadcast, host promoted
    23‚Üí    23‚Üí *      (first remaining by Map insertion order), active vote scrubbed,
    24‚Üí    24‚Üí *      "left" activity logged.
    25‚Üí    25‚Üí *
    26‚Üí    26‚Üí * seq numbers: every state-changing broadcast (board:update relay, vote:state,
    27‚Üí    27‚Üí * vote:result, presence:update, activity:new) increments `room.seq` and
    28‚Üí    28‚Üí * includes the new seq in its payload. Clients use this as a monotonic
    29‚Üí    29‚Üí * ordering signal for reconciliation / gap detection.
    30‚Üí    30‚Üí *
    31‚Üí    31‚Üí * Idempotency: `board:update` carries an optional `eventId`. The server does
    32‚Üí    32‚Üí * NOT dedupe by eventId (full-board sync is inherently idempotent), but it
    33‚Üí    33‚Üí * relays `{ board, seq, eventId }` so clients can skip their own echo
    34‚Üí    34‚Üí * (match eventId) and detect gaps (seq).
    35‚Üí    35‚Üí */

   148‚Üí   148‚Üí/** Generate a unique id (crypto.randomUUID with safe fallback). */

   150‚Üí   150‚Üí  try {
   151‚Üí   151‚Üí    if (typeof crypto !== 'undefined' && typeof crypto.randomUUID === 'function') {
   152‚Üí   152‚Üí      return crypto.randomUUID()
   153‚Üí   153‚Üí    }
   154‚Üí   154‚Üí  } catch {
   155‚Üí   155‚Üí    /* fall through */
   156‚Üí   156‚Üí  }
   157‚Üí   157‚Üí  return Math.random().toString(36).slice(2) + Date.now().toString(36)
   158‚Üí   158‚Üí}

   195‚Üí   195‚Üí/** Track which rooms a socket has joined (for fast disconnect cleanup). */
   196‚Üí   196‚Üífunction rememberRoom(socketId: string, roomId: string): void {
   197‚Üí   197‚Üí  let set = socketRooms.get(socketId)
   198‚Üí   198‚Üí  if (!set) {
   199‚Üí   199‚Üí    set = new Set<string>()
   200‚Üí   200‚Üí    socketRooms.set(socketId, set)
   201‚Üí   201‚Üí  }
   202‚Üí   202‚Üí  set.add(roomId)
   203‚Üí   203‚Üí}
   204‚Üí   204‚Üí
   205‚Üí   205‚Üí/**
   206‚Üí   206‚Üí * Remove a socket from a room and clean up the reverse index. If the host
   207‚Üí   207‚Üí * left, promote the first remaining member by Map insertion order (which is
   208‚Üí   208‚Üí * the join order). Does NOT broadcast ‚Äî the caller is responsible for that.
   209‚Üí   209‚Üí */
   210‚Üí   210‚Üífunction forgetRoom(socketId: string, roomId: string): void {
   211‚Üí   211‚Üí  const room = rooms.get(roomId)
   212‚Üí   212‚Üí  if (room) {
   213‚Üí   213‚Üí    room.members.delete(socketId)
   214‚Üí   214‚Üí    if (room.host === socketId) {
   215‚Üí   215‚Üí      // Promote the first remaining member by insertion order.
   216‚Üí   216‚Üí      const next = room.members.size > 0 ? (room.members.keys().next().value ?? null) : null
   217‚Üí   217‚Üí      room.host = next
   218‚Üí   218‚Üí    }
   219‚Üí   219‚Üí    // If empty, we keep the snapshot in memory so a quick reconnect restores
   220‚Üí   220‚Üí    // state. The room Map entry stays around; that's fine for a mini-service.
   221‚Üí   221‚Üí  }
   222‚Üí   222‚Üí  socketRooms.get(socketId)?.delete(roomId)
   223‚Üí   223‚Üí}

   235‚Üí   235‚Üí/** Push an activity entry and cap the stored log at MAX_ACTIVITY. */

   281‚Üí   281‚Üí/** Build the vote:state payload to broadcast. tally is { tierId: count }. */
   282‚Üí   282‚Üífunction voteStatePayload(room: RoomState, seq?: number): {
   283‚Üí   283‚Üí  active: boolean
   284‚Üí   284‚Üí  itemId: string | null
   285‚Üí   285‚Üí  item: Record<string, unknown> | null
   286‚Üí   286‚Üí  tally: Record<string, number>
   287‚Üí   287‚Üí  voterCount: number
   288‚Üí   288‚Üí  totalPeers: number
   289‚Üí   289‚Üí  seq: number
   290‚Üí   290‚Üí} {
   291‚Üí   291‚Üí  const vote = room.vote
   292‚Üí   292‚Üí  const tally: Record<string, number> = {}
   293‚Üí   293‚Üí  let voterCount = 0
   294‚Üí   294‚Üí  if (vote) {
   295‚Üí   295‚Üí    for (const tierId of vote.votes.values()) {
   296‚Üí   296‚Üí      tally[tierId] = (tally[tierId] ?? 0) + 1
   297‚Üí   297‚Üí      voterCount++
   298‚Üí   298‚Üí    }
   299‚Üí   299‚Üí  }
   300‚Üí   300‚Üí  return {
   301‚Üí   301‚Üí    active: !!vote,
   302‚Üí   302‚Üí    itemId: vote?.itemId ?? null,
   303‚Üí   303‚Üí    item: vote?.item ?? null,
   304‚Üí   304‚Üí    tally,
   305‚Üí   305‚Üí    voterCount,
   306‚Üí   306‚Üí    totalPeers: room.members.size,
   307‚Üí   307‚Üí    seq: seq ?? room.seq,
   308‚Üí   308‚Üí  }
   309‚Üí   309‚Üí}

   324‚Üí   324‚Üí  // so the socket.io path MUST be '/' to match the default.
   325‚Üí   325‚Üí  path: '/',
   326‚Üí   326‚Üí  cors: {
   327‚Üí   327‚Üí    origin: '*',
   328‚Üí   328‚Üí    methods: ['GET', 'POST'],

   336‚Üí   336‚Üí// ---------------------------------------------------------------------------
   337‚Üí   337‚Üí
   338‚Üí   338‚Üí/**
   339‚Üí   339‚Üí * Every HEARTBEAT_SWEEP_MS, iterate all rooms and force-disconnect any member
   340‚Üí   340‚Üí * whose `lastSeen` is older than HEARTBEAT_TIMEOUT_MS. The normal `disconnect`
   341‚Üí   341‚Üí * handler then takes care of cleanup (presence, host promotion, vote scrub,
   342‚Üí   342‚Üí * activity log). This reclaims dead sessions that the socket.io ping/pong
   343‚Üí   343‚Üí * layer might miss (e.g. half-open TCP).
   344‚Üí   344‚Üí */
   345‚Üí   345‚Üíconst heartbeatSweep = setInterval(() => {
   346‚Üí   346‚Üí  try {
   347‚Üí   347‚Üí    const now = Date.now()
   348‚Üí   348‚Üí    for (const [roomId, room] of rooms) {
   349‚Üí   349‚Üí      for (const [socketId, user] of room.members) {
   350‚Üí   350‚Üí        const idle = now - user.lastSeen
   351‚Üí   351‚Üí        if (idle > HEARTBEAT_TIMEOUT_MS) {
   352‚Üí   352‚Üí          const s = io.sockets.sockets.get(socketId)
   353‚Üí   353‚Üí          if (s && s.connected) {
   354‚Üí   354‚Üí            console.log(
   355‚Üí   355‚Üí              `[heartbeat] force-disconnect ${socketId} in ${roomId} (idle ${idle}ms)`,
   356‚Üí   356‚Üí            )
   357‚Üí   357‚Üí            // `true` = close the underlying connection. This triggers the
   358‚Üí   358‚Üí            // normal `disconnect` handler, which cleans up state.
   359‚Üí   359‚Üí            s.disconnect(true)
   360‚Üí   360‚Üí          }
   361‚Üí   361‚Üí        }
   362‚Üí   362‚Üí      }
   363‚Üí   363‚Üí    }
   364‚Üí   364‚Üí  } catch (err) {
   365‚Üí   365‚Üí    console.error('[heartbeat] sweep error:', err)
   366‚Üí   366‚Üí  }
   367‚Üí   367‚Üí}, HEARTBEAT_SWEEP_MS)
   368‚Üí   368‚Üí
   369‚Üí   369‚Üí// ---------------------------------------------------------------------------
   370‚Üí   370‚Üí// Connection lifecycle
   371‚Üí   371‚Üí// ---------------------------------------------------------------------------
   372‚Üí   372‚Üí
   373‚Üí   373‚Üíio.on('connection', (socket: Socket) => {
   374‚Üí   374‚Üí  console.log(`[connect] ${socket.id}`)
   375‚Üí   375‚Üí
   376‚Üí   376‚Üí  // ----- identity --------------------------------------------------------
   377‚Üí   377‚Üí  // Set display name + color on the socket. Does NOT join a room yet ‚Äî the
   378‚Üí   378‚Üí  // client must follow up with `room:join`. Until then the socket is a
   379‚Üí   379‚Üí  // "pending" socket (connected but not a member of any room).
   380‚Üí   380‚Üí  socket.on('identity', (payload: unknown) => {
   381‚Üí   381‚Üí    try {
   382‚Üí   382‚Üí      const data = (payload ?? {}) as { name?: string; color?: string }
   383‚Üí   383‚Üí      let name = typeof data.name === 'string' ? data.name.trim().slice(0, 20) : ''
   384‚Üí   384‚Üí      if (!name) name = 'Guest'
   385‚Üí   385‚Üí      const color =
   386‚Üí   386‚Üí        typeof data.color === 'string' && /^#[0-9a-fA-F]{6}$/.test(data.color)
   387‚Üí   387‚Üí          ? data.color
   388‚Üí   388‚Üí          : '#64748b'
   389‚Üí   389‚Üí      socket.data.identity = { name, color }
   390‚Üí   390‚Üí      console.log(`[identity] ${socket.id} name=${name} color=${color}`)
   391‚Üí   391‚Üí    } catch (err) {
   392‚Üí   392‚Üí      console.error(`[identity] error from ${socket.id}:`, err)
   393‚Üí   393‚Üí    }
   394‚Üí   394‚Üí  })
   395‚Üí   395‚Üí
   396‚Üí   396‚Üí  // ----- room:join -------------------------------------------------------
   397‚Üí   397‚Üí  // Join (or create) a room. Enforces the 10-user cap. Reads identity from
   398‚Üí   398‚Üí  // socket.data (set via `identity` event; defaults to Guest/#64748b). Sends
   399‚Üí   399‚Üí  // the joiner a full state bundle (room:state + presence:update +
   400‚Üí   400‚Üí  // activity:sync + vote:state if active), then broadcasts presence + a
   401‚Üí   401‚Üí  // "joined" activity to the room.
   402‚Üí   402‚Üí  socket.on('room:join', (payload: unknown) => {
   403‚Üí   403‚Üí    try {
   404‚Üí   404‚Üí      const data = (payload ?? {}) as { roomId?: string; board?: Board }
   405‚Üí   405‚Üí      const roomId = typeof data.roomId === 'string' ? data.roomId.trim() : ''
   406‚Üí   406‚Üí      if (!roomId) {
   407‚Üí   407‚Üí        socket.emit('room:error', { event: 'room:join', message: 'roomId is required' })
   408‚Üí   408‚Üí        return
   409‚Üí   409‚Üí      }
   410‚Üí   410‚Üí
   411‚Üí   411‚Üí      // 10-user cap: reject if the room is already full.
   412‚Üí   412‚Üí      const existing = rooms.get(roomId)
   413‚Üí   413‚Üí      if (existing && existing.members.size >= MAX_MEMBERS) {
   414‚Üí   414‚Üí        socket.emit('room:error', {
   415‚Üí   415‚Üí          event: 'room:join',
   416‚Üí   416‚Üí          message: 'Room is full (max 10)',
   417‚Üí   417‚Üí        })
   418‚Üí   418‚Üí        return
   419‚Üí   419‚Üí      }
   420‚Üí   420‚Üí
   421‚Üí   421‚Üí      // Leave any rooms this socket was previously in (a socket is in one
   422‚Üí   422‚Üí      // RankForge room at a time). Broadcast presence:update + scrub any
   423‚Üí   423‚Üí      // active vote from the old room.
   424‚Üí   424‚Üí      const previous = socketRooms.get(socket.id)
   425‚Üí   425‚Üí      if (previous && previous.size > 0) {
   426‚Üí   426‚Üí        for (const oldRoomId of previous) {
   427‚Üí   427‚Üí          const oldRoom = rooms.get(oldRoomId)
   428‚Üí   428‚Üí          forgetRoom(socket.id, oldRoomId)
   429‚Üí   429‚Üí          socket.leave(oldRoomId)
   430‚Üí   430‚Üí          if (oldRoom) {
   431‚Üí   431‚Üí            broadcastPresence(io, oldRoom, oldRoomId)
   432‚Üí   432‚Üí            // If the switching user had voted on an active vote, drop it
   433‚Üí   433‚Üí            // and re-broadcast the tally so the count is live.
   434‚Üí   434‚Üí            if (oldRoom.vote && oldRoom.vote.votes.has(socket.id)) {
   435‚Üí   435‚Üí              oldRoom.vote.votes.delete(socket.id)
   436‚Üí   436‚Üí              io.to(oldRoomId).emit('vote:state', voteStatePayload(oldRoom, nextSeq(oldRoom)))
   437‚Üí   437‚Üí            }
   438‚Üí   438‚Üí          }
   439‚Üí   439‚Üí        }
   440‚Üí   440‚Üí      }
   441‚Üí   441‚Üí
   442‚Üí   442‚Üí      const room = getOrCreateRoom(roomId)
   443‚Üí   443‚Üí      const wasNew = room.members.size === 0
   444‚Üí   444‚Üí
   445‚Üí   445‚Üí      // Read identity (set via `identity` event; default to Guest/#64748b).
   446‚Üí   446‚Üí      const identity =
   447‚Üí   447‚Üí        (socket.data?.identity as { name: string; color: string } | undefined) ?? {
   448‚Üí   448‚Üí          name: 'Guest',
   449‚Üí   449‚Üí          color: '#64748b',
   450‚Üí   450‚Üí        }
   451‚Üí   451‚Üí
   452‚Üí   452‚Üí      const user: User = {
   453‚Üí   453‚Üí        id: socket.id,
   454‚Üí   454‚Üí        name: identity.name,
   455‚Üí   455‚Üí        color: identity.color,
   456‚Üí   456‚Üí        presence: 'online',
   457‚Üí   457‚Üí        lastSeen: Date.now(),
   458‚Üí   458‚Üí      }
   459‚Üí   459‚Üí
   460‚Üí   460‚Üí      socket.join(roomId)
   461‚Üí   461‚Üí      room.members.set(socket.id, user)
   462‚Üí   462‚Üí      rememberRoom(socket.id, roomId)
   463‚Üí   463‚Üí
   464‚Üí   464‚Üí      // First joiner (or recovered host-less room) becomes host.
   465‚Üí   465‚Üí      if (wasNew || room.host === null) {
   466‚Üí   466‚Üí        room.host = socket.id
   467‚Üí   467‚Üí      }
   468‚Üí   468‚Üí
   469‚Üí   469‚Üí      // Update the stored snapshot if the client provided a board.
   470‚Üí   470‚Üí      if (data.board !== undefined && data.board !== null) {
   471‚Üí   471‚Üí        room.board = data.board
   472‚Üí   472‚Üí      }
   473‚Üí   473‚Üí
   474‚Üí   474‚Üí      const isHost = room.host === socket.id
   475‚Üí   475‚Üí
   476‚Üí   476‚Üí      // 1. Full state bundle to the joiner.
   477‚Üí   477‚Üí      socket.emit('room:state', {
   478‚Üí   478‚Üí        roomId,
   479‚Üí   479‚Üí        isHost,
   480‚Üí   480‚Üí        peers: room.members.size,
   481‚Üí   481‚Üí        board: room.board,
   482‚Üí   482‚Üí        seq: room.seq,
   483‚Üí   483‚Üí      })
   484‚Üí   484‚Üí
   485‚Üí   485‚Üí      // 2. Presence to the joiner (direct; uses current seq, no increment).
   486‚Üí   486‚Üí      socket.emit('presence:update', {
   487‚Üí   487‚Üí        members: presenceList(room),
   488‚Üí   488‚Üí        host: room.host,
   489‚Üí   489‚Üí        seq: room.seq,
   490‚Üí   490‚Üí      })
   491‚Üí   491‚Üí
   492‚Üí   492‚Üí      // 3. Activity log to the joiner (last 40 entries).
   493‚Üí   493‚Üí      socket.emit('activity:sync', { entries: activityPayload(room) })
   494‚Üí   494‚Üí
   495‚Üí   495‚Üí      // 4. If a vote is currently active, hydrate the joiner (no seq bump).
   496‚Üí   496‚Üí      if (room.vote) {
   497‚Üí   497‚Üí        socket.emit('vote:state', voteStatePayload(room))
   498‚Üí   498‚Üí      }
   499‚Üí   499‚Üí
   500‚Üí   500‚Üí      // 5. Broadcast presence to the whole room (including joiner).
   501‚Üí   501‚Üí      broadcastPresence(io, room, roomId)
   502‚Üí   502‚Üí
   503‚Üí   503‚Üí      // 6. Push + broadcast "joined" activity to the whole room.
   504‚Üí   504‚Üí      broadcastActivity(io, room, roomId, {
   505‚Üí   505‚Üí        id: uuid(),
   506‚Üí   506‚Üí        userId: socket.id,
   507‚Üí   507‚Üí        userName: user.name,
   508‚Üí   508‚Üí        action: 'joined',
   509‚Üí   509‚Üí        detail: user.name,
   510‚Üí   510‚Üí        ts: Date.now(),
   511‚Üí   511‚Üí      })
   512‚Üí   512‚Üí
   513‚Üí   513‚Üí      console.log(
   514‚Üí   514‚Üí        `[room:join] ${socket.id} -> ${roomId} (peers=${room.members.size}, host=${isHost})`,
   515‚Üí   515‚Üí      )
   516‚Üí   516‚Üí    } catch (err) {
   517‚Üí   517‚Üí      console.error(`[room:join] error from ${socket.id}:`, err)
   518‚Üí   518‚Üí      socket.emit('room:error', { event: 'room:join', message: 'internal error' })
   519‚Üí   519‚Üí    }
   520‚Üí   520‚Üí  })
   521‚Üí   521‚Üí
   522‚Üí   522‚Üí  // ----- board:update ----------------------------------------------------
   523‚Üí   523‚Üí  // A client changed its local board and wants to broadcast to peers.
   524‚Üí   524‚Üí  // We store the snapshot AND relay to everyone else (NOT back to sender,
   525‚Üí   525‚Üí  // since the sender already has the optimistic local state). Includes the
   526‚Üí   526‚Üí  // new seq + the client-supplied eventId so peers can skip re-applying their
   527‚Üí   527‚Üí  // own echo and detect gaps.
   528‚Üí   528‚Üí  socket.on('board:update', (payload: unknown) => {
   529‚Üí   529‚Üí    try {
   530‚Üí   530‚Üí      const data = (payload ?? {}) as {
   531‚Üí   531‚Üí        roomId?: string
   532‚Üí   532‚Üí        board?: Board
   533‚Üí   533‚Üí        eventId?: string
   534‚Üí   534‚Üí      }
   535‚Üí   535‚Üí      const roomId = typeof data.roomId === 'string' ? data.roomId : ''
   536‚Üí   536‚Üí      if (!roomId || data.board === undefined || data.board === null) return
   537‚Üí   537‚Üí
   538‚Üí   538‚Üí      const room = rooms.get(roomId)
   539‚Üí   539‚Üí      if (!room) return // must join before updating
   540‚Üí   540‚Üí
   541‚Üí   541‚Üí      room.board = data.board
   542‚Üí   542‚Üí      const seq = nextSeq(room)
   543‚Üí   543‚Üí      // Relay to OTHERS only (sender already has optimistic state).
   544‚Üí   544‚Üí      socket.to(roomId).emit('board:update', {
   545‚Üí   545‚Üí        board: data.board,
   546‚Üí   546‚Üí        seq,
   547‚Üí   547‚Üí        eventId: typeof data.eventId === 'string' ? data.eventId : undefined,
   548‚Üí   548‚Üí      })
   549‚Üí   549‚Üí    } catch (err) {
   550‚Üí   550‚Üí      console.error(`[board:update] error from ${socket.id}:`, err)
   551‚Üí   551‚Üí    }
   552‚Üí   552‚Üí  })
   553‚Üí   553‚Üí
   554‚Üí   554‚Üí  // ----- activity:log ----------------------------------------------------
   555‚Üí   555‚Üí  // Client-driven activity logging. The sender emits this right after a
   556‚Üí   556‚Üí  // user-visible board action (add/move/delete) so the activity feed can say
   557‚Üí   557‚Üí  // "Mason moved Street Tacos to A" without the server parsing board diffs
   558‚Üí   558‚Üí  // (which is fragile). The server just validates, builds the entry, and
   559‚Üí   559‚Üí  // broadcasts it to the whole room (including sender).
   560‚Üí   560‚Üí  socket.on('activity:log', (payload: unknown) => {
   561‚Üí   561‚Üí    try {
   562‚Üí   562‚Üí      const data = (payload ?? {}) as { roomId?: string; action?: string; detail?: string }
   563‚Üí   563‚Üí      const roomId = typeof data.roomId === 'string' ? data.roomId.trim() : ''
   564‚Üí   564‚Üí      const action = typeof data.action === 'string' ? data.action : ''
   565‚Üí   565‚Üí      const detail =
   566‚Üí   566‚Üí        typeof data.detail === 'string' ? data.detail.trim().slice(0, 120) : ''
   567‚Üí   567‚Üí      if (!roomId || !ALLOWED_ACTIONS.has(action)) return
   568‚Üí   568‚Üí
   569‚Üí   569‚Üí      const room = rooms.get(roomId)
   570‚Üí   570‚Üí      if (!room) return
   571‚Üí   571‚Üí      const user = room.members.get(socket.id)
   572‚Üí   572‚Üí      if (!user) return
   573‚Üí   573‚Üí
   574‚Üí   574‚Üí      const entry: ActivityEntry = {
   575‚Üí   575‚Üí        id: uuid(),
   576‚Üí   576‚Üí        userId: socket.id,
   577‚Üí   577‚Üí        userName: user.name,
   578‚Üí   578‚Üí        action: action as ActivityEntry['action'],
   579‚Üí   579‚Üí        detail,
   580‚Üí   580‚Üí        ts: Date.now(),
   581‚Üí   581‚Üí      }
   582‚Üí   582‚Üí      broadcastActivity(io, room, roomId, entry)
   583‚Üí   583‚Üí      console.log(`[activity:log] ${socket.id} ${action} "${detail}" in ${roomId}`)
   584‚Üí   584‚Üí    } catch (err) {
   585‚Üí   585‚Üí      console.error(`[activity:log] error from ${socket.id}:`, err)
   586‚Üí   586‚Üí    }
   587‚Üí   587‚Üí  })
   588‚Üí   588‚Üí
   589‚Üí   589‚Üí  // ----- presence:set ----------------------------------------------------
   590‚Üí   590‚Üí  // Client tells the server its current fine-grained presence state (online /
   591‚Üí   591‚Üí  // idle / dragging / voting). Clients should throttle this on their side.
   592‚Üí   592‚Üí  // Server updates the user and re-broadcasts presence to the room.
   593‚Üí   593‚Üí  socket.on('presence:set', (payload: unknown) => {
   594‚Üí   594‚Üí    try {
   595‚Üí   595‚Üí      const data = (payload ?? {}) as { roomId?: string; state?: string }
   596‚Üí   596‚Üí      const roomId = typeof data.roomId === 'string' ? data.roomId.trim() : ''
   597‚Üí   597‚Üí      const state = typeof data.state === 'string' ? data.state : ''
   598‚Üí   598‚Üí      if (!roomId || !PRESENCE_STATES.has(state)) return
   599‚Üí   599‚Üí
   600‚Üí   600‚Üí      const room = rooms.get(roomId)
   601‚Üí   601‚Üí      if (!room) return
   602‚Üí   602‚Üí      const user = room.members.get(socket.id)
   603‚Üí   603‚Üí      if (!user) return
   604‚Üí   604‚Üí
   605‚Üí   605‚Üí      user.presence = state as User['presence']
   606‚Üí   606‚Üí      user.lastSeen = Date.now()
   607‚Üí   607‚Üí      broadcastPresence(io, room, roomId)
   608‚Üí   608‚Üí    } catch (err) {
   609‚Üí   609‚Üí      console.error(`[presence:set] error from ${socket.id}:`, err)
   610‚Üí   610‚Üí    }
   611‚Üí   611‚Üí  })
   612‚Üí   612‚Üí
   613‚Üí   613‚Üí  // ----- focus:set -------------------------------------------------------
   614‚Üí   614‚Üí  // A user is hovering / started dragging a specific item. PURE RELAY ‚Äî we
   615‚Üí   615‚Üí  // do NOT store any focus state server-side (it's ephemeral; clients clear
   616‚Üí   616‚Üí  // it on a timer). Validates roomId + itemId are non-empty strings, looks
   617‚Üí   617‚Üí  // up the room + the user (socket.id -> members; if not found, silent
   618‚Üí   618‚Üí  // return), then relays to OTHER clients in the room so they can render a
   619‚Üí   619‚Üí  // colored highlight + the focusing user's avatar on that item.
   620‚Üí   620‚Üí  // High-frequency event: NO log line per emit (would spam ‚Äî debug only if
   621‚Üí   621‚Üí  // ever needed).
   622‚Üí   622‚Üí  socket.on('focus:set', (payload: unknown) => {
   623‚Üí   623‚Üí    try {
   624‚Üí   624‚Üí      const data = (payload ?? {}) as { roomId?: string; itemId?: string }
   625‚Üí   625‚Üí      const roomId = typeof data.roomId === 'string' ? data.roomId.trim() : ''
   626‚Üí   626‚Üí      const itemId = typeof data.itemId === 'string' ? data.itemId.trim() : ''
   627‚Üí   627‚Üí      if (!roomId || !itemId) return
   628‚Üí   628‚Üí
   629‚Üí   629‚Üí      const room = rooms.get(roomId)
   630‚Üí   630‚Üí      if (!room) return
   631‚Üí   631‚Üí      const user = room.members.get(socket.id)
   632‚Üí   632‚Üí      if (!user) return
   633‚Üí   633‚Üí
   634‚Üí   634‚Üí      socket.to(roomId).emit('focus:set', {
   635‚Üí   635‚Üí        userId: socket.id,
   636‚Üí   636‚Üí        userName: user.name,
   637‚Üí   637‚Üí        userColor: user.color,
   638‚Üí   638‚Üí        itemId,
   639‚Üí   639‚Üí      })
   640‚Üí   640‚Üí    } catch (err) {
   641‚Üí   641‚Üí      console.error(`[focus:set] error from ${socket.id}:`, err)
   642‚Üí   642‚Üí    }
   643‚Üí   643‚Üí  })
   644‚Üí   644‚Üí
   645‚Üí   645‚Üí  // ----- focus:clear -----------------------------------------------------
   646‚Üí   646‚Üí  // A user stopped hovering / dragging. PURE RELAY ‚Äî no server state to
   647‚Üí   647‚Üí  // clean up. Validates roomId, then relays to OTHER clients in the room so
   648‚Üí   648‚Üí  // they can drop that user's highlight.
   649‚Üí   649‚Üí  socket.on('focus:clear', (payload: unknown) => {
   650‚Üí   650‚Üí    try {
   651‚Üí   651‚Üí      const data = (payload ?? {}) as { roomId?: string }
   652‚Üí   652‚Üí      const roomId = typeof data.roomId === 'string' ? data.roomId.trim() : ''
   653‚Üí   653‚Üí      if (!roomId) return
   654‚Üí   654‚Üí
   655‚Üí   655‚Üí      socket.to(roomId).emit('focus:clear', { userId: socket.id })
   656‚Üí   656‚Üí    } catch (err) {
   657‚Üí   657‚Üí      console.error(`[focus:clear] error from ${socket.id}:`, err)
   658‚Üí   658‚Üí    }
   659‚Üí   659‚Üí  })
   660‚Üí   660‚Üí
   661‚Üí   661‚Üí  // ----- heartbeat -------------------------------------------------------
   662‚Üí   662‚Üí  // Client emits this every ~20s (no payload). Server refreshes the user's
   663‚Üí   663‚Üí  // lastSeen across all rooms they're in. The server-side sweep uses
   664‚Üí   664‚Üí  // lastSeen to decide who to force-disconnect.
   665‚Üí   665‚Üí  socket.on('heartbeat', () => {
   666‚Üí   666‚Üí    try {
   667‚Üí   667‚Üí      const joined = socketRooms.get(socket.id)
   668‚Üí   668‚Üí      if (!joined || joined.size === 0) return
   669‚Üí   669‚Üí      const now = Date.now()
   670‚Üí   670‚Üí      for (const roomId of joined) {
   671‚Üí   671‚Üí        const room = rooms.get(roomId)
   672‚Üí   672‚Üí        const user = room?.members.get(socket.id)
   673‚Üí   673‚Üí        if (user) user.lastSeen = now
   674‚Üí   674‚Üí      }
   675‚Üí   675‚Üí    } catch (err) {
   676‚Üí   676‚Üí      console.error(`[heartbeat] error from ${socket.id}:`, err)
   677‚Üí   677‚Üí    }
   678‚Üí   678‚Üí  })
   679‚Üí   679‚Üí
   680‚Üí   680‚Üí  // ----- board:sync-request ---------------------------------------------
   681‚Üí   681‚Üí  // A freshly-joined client asks for the current board. We reply directly

   683‚Üí   683‚Üí  socket.on('board:sync-request', (payload: unknown) => {

   691‚Üí   691‚Üí        board: room ? room.board : null,

   693‚Üí   693‚Üí    } catch (err) {

   696‚Üí   696‚Üí  })
   697‚Üí   697‚Üí

   716‚Üí   716‚Üí              voterCount: 0,
   717‚Üí   717‚Üí              totalPeers: 0,
   718‚Üí   718‚Üí              seq: 0,
   719‚Üí   719‚Üí            },
   720‚Üí   720‚Üí      )

   724‚Üí   724‚Üí  })
   725‚Üí   725‚Üí
   726‚Üí   726‚Üí  // ----- vote:start ------------------------------------------------------
   727‚Üí   727‚Üí  // Host kicks off a vote on a specific item. The item "pops up" for everyone
   728‚Üí   728‚Üí  // in the room; each peer then casts one vote for a tier via `vote:cast`.
   729‚Üí   729‚Üí  // Also logs a "vote_started" activity entry.
   730‚Üí   730‚Üí  socket.on('vote:start', (payload: unknown) => {
   731‚Üí   731‚Üí    try {
   732‚Üí   732‚Üí      const data = (payload ?? {}) as {
   733‚Üí   733‚Üí        roomId?: string
   734‚Üí   734‚Üí        itemId?: string
   735‚Üí   735‚Üí        item?: Record<string, unknown>
   736‚Üí   736‚Üí      }
   737‚Üí   737‚Üí      const roomId = typeof data.roomId === 'string' ? data.roomId.trim() : ''
   738‚Üí   738‚Üí      const itemId = typeof data.itemId === 'string' ? data.itemId.trim() : ''
   739‚Üí   739‚Üí      if (
   740‚Üí   740‚Üí        !roomId ||
   741‚Üí   741‚Üí        !itemId ||
   742‚Üí   742‚Üí        typeof data.item !== 'object' ||
   743‚Üí   743‚Üí        data.item === null ||
   744‚Üí   744‚Üí        Array.isArray(data.item)
   745‚Üí   745‚Üí      ) {
   746‚Üí   746‚Üí        socket.emit('room:error', {
   747‚Üí   747‚Üí          event: 'vote:start',
   748‚Üí   748‚Üí          message: 'roomId (string), itemId (string) and item (object) are required',
   749‚Üí   749‚Üí        })
   750‚Üí   750‚Üí        return
   751‚Üí   751‚Üí      }
   752‚Üí   752‚Üí
   753‚Üí   753‚Üí      const room = rooms.get(roomId)
   754‚Üí   754‚Üí      if (!room) {
   755‚Üí   755‚Üí        socket.emit('room:error', { event: 'vote:start', message: 'room not found' })
   756‚Üí   756‚Üí        return
   757‚Üí   757‚Üí      }
   758‚Üí   758‚Üí      if (room.host !== socket.id) {
   759‚Üí   759‚Üí        socket.emit('room:error', { event: 'vote:start', message: 'only host can start a vote' })
   760‚Üí   760‚Üí        return
   761‚Üí   761‚Üí      }
   762‚Üí   762‚Üí
   763‚Üí   763‚Üí      room.vote = {
   764‚Üí   764‚Üí        itemId,
   765‚Üí   765‚Üí        item: data.item,
   766‚Üí   766‚Üí        votes: new Map<string, string>(),
   767‚Üí   767‚Üí        startedBy: socket.id,
   768‚Üí   768‚Üí      }
   769‚Üí   769‚Üí
   770‚Üí   770‚Üí      io.to(roomId).emit('vote:state', voteStatePayload(room, nextSeq(room)))
   771‚Üí   771‚Üí
   772‚Üí   772‚Üí      // NOTE: vote_started activity is sent by the client via activity:log
   773‚Üí   773‚Üí      // (the client has the readable item label; we don't duplicate it here).
   774‚Üí   774‚Üí
   775‚Üí   775‚Üí      console.log(`[vote:start] ${socket.id} -> ${roomId} item=${itemId}`)
   776‚Üí   776‚Üí    } catch (err) {
   777‚Üí   777‚Üí      console.error(`[vote:start] error from ${socket.id}:`, err)
   778‚Üí   778‚Üí      socket.emit('room:error', { event: 'vote:start', message: 'internal error' })
   779‚Üí   779‚Üí    }
   780‚Üí   780‚Üí  })
   781‚Üí   781‚Üí
   782‚Üí   782‚Üí  // ----- vote:cast --------------------------------------------------------
   783‚Üí   783‚Üí  // A peer casts (or re-casts) their vote for a tier on the active item.
   784‚Üí   784‚Üí  // Stale votes (wrong itemId / no active vote) are ignored silently.
   785‚Üí   785‚Üí  // Also logs a "voted" activity entry (detail = "<item label> -> <tierId>").
   786‚Üí   786‚Üí  socket.on('vote:cast', (payload: unknown) => {
   787‚Üí   787‚Üí    try {
   788‚Üí   788‚Üí      const data = (payload ?? {}) as {
   789‚Üí   789‚Üí        roomId?: string
   790‚Üí   790‚Üí        itemId?: string
   791‚Üí   791‚Üí        tierId?: string
   792‚Üí   792‚Üí      }
   793‚Üí   793‚Üí      const roomId = typeof data.roomId === 'string' ? data.roomId.trim() : ''
   794‚Üí   794‚Üí      const itemId = typeof data.itemId === 'string' ? data.itemId.trim() : ''
   795‚Üí   795‚Üí      const tierId = typeof data.tierId === 'string' ? data.tierId.trim() : ''
   796‚Üí   796‚Üí      if (!roomId || !itemId || !tierId) return
   797‚Üí   797‚Üí
   798‚Üí   798‚Üí      const room = rooms.get(roomId)
   799‚Üí   799‚Üí      if (!room || !room.vote) return
   800‚Üí   800‚Üí      if (room.vote.itemId !== itemId) return // stale vote
   801‚Üí   801‚Üí
   802‚Üí   802‚Üí      room.vote.votes.set(socket.id, tierId)
   803‚Üí   803‚Üí      io.to(roomId).emit('vote:state', voteStatePayload(room, nextSeq(room)))
   804‚Üí   804‚Üí
   805‚Üí   805‚Üí      // NOTE: voted activity is sent by the client via activity:log
   806‚Üí   806‚Üí      // (the client formats it as "S for Arcade Night" which is more readable).
   807‚Üí   807‚Üí
   808‚Üí   808‚Üí      console.log(`[vote:cast] ${socket.id} voted ${tierId} in ${roomId}`)
   809‚Üí   809‚Üí    } catch (err) {
   810‚Üí   810‚Üí      console.error(`[vote:cast] error from ${socket.id}:`, err)
   811‚Üí   811‚Üí    }
   812‚Üí   812‚Üí  })
   813‚Üí   813‚Üí
   814‚Üí   814‚Üí  // ----- vote:end ---------------------------------------------------------
   815‚Üí   815‚Üí  // Host ends the active vote. Winner = tierId with the highest count; ties
   816‚Üí   816‚Üí  // broken by ascending alphabetical tierId (deterministic). If no votes were
   817‚Üí   817‚Üí  // cast, winner = null. Emits `vote:result` then a final inactive `vote:state`
   818‚Üí   818‚Üí  // so clients close the overlay. Also logs a "vote_ended" activity.
   819‚Üí   819‚Üí  socket.on('vote:end', (payload: unknown) => {
   820‚Üí   820‚Üí    try {
   821‚Üí   821‚Üí      const data = (payload ?? {}) as { roomId?: string }
   822‚Üí   822‚Üí      const roomId = typeof data.roomId === 'string' ? data.roomId.trim() : ''
   823‚Üí   823‚Üí      if (!roomId) return
   824‚Üí   824‚Üí
   825‚Üí   825‚Üí      const room = rooms.get(roomId)
   826‚Üí   826‚Üí      if (!room || !room.vote) return
   827‚Üí   827‚Üí
   828‚Üí   828‚Üí      if (room.host !== socket.id) {
   829‚Üí   829‚Üí        socket.emit('room:error', { event: 'vote:end', message: 'only host can end a vote' })
   830‚Üí   830‚Üí        return
   831‚Üí   831‚Üí      }
   832‚Üí   832‚Üí
   833‚Üí   833‚Üí      const vote = room.vote
   834‚Üí   834‚Üí      const tally: Record<string, number> = {}
   835‚Üí   835‚Üí      let winner: string | null = null
   836‚Üí   836‚Üí      let bestCount = -1
   837‚Üí   837‚Üí      // Iterating tierIds in sorted order gives deterministic tie-breaking:
   838‚Üí   838‚Üí      // the alphabetically-first tierId wins on a tie (strict > keeps the
   839‚Üí   839‚Üí      // earliest-seen leading candidate).
   840‚Üí   840‚Üí      const sortedTierIds = Array.from(vote.votes.values()).sort()
   841‚Üí   841‚Üí      for (const tierId of sortedTierIds) {
   842‚Üí   842‚Üí        const count = (tally[tierId] ?? 0) + 1
   843‚Üí   843‚Üí        tally[tierId] = count
   844‚Üí   844‚Üí        if (count > bestCount) {
   845‚Üí   845‚Üí          bestCount = count
   846‚Üí   846‚Üí          winner = tierId
   847‚Üí   847‚Üí        }
   848‚Üí   848‚Üí      }
   849‚Üí   849‚Üí      if (vote.votes.size === 0) winner = null
   850‚Üí   850‚Üí
   851‚Üí   851‚Üí      io.to(roomId).emit('vote:result', {
   852‚Üí   852‚Üí        itemId: vote.itemId,
   853‚Üí   853‚Üí        tally,
   854‚Üí   854‚Üí        winner,
   855‚Üí   855‚Üí        seq: nextSeq(room),
   856‚Üí   856‚Üí      })
   857‚Üí   857‚Üí
   858‚Üí   858‚Üí      room.vote = null
   859‚Üí   859‚Üí      io.to(roomId).emit('vote:state', voteStatePayload(room, nextSeq(room)))
   860‚Üí   860‚Üí
   861‚Üí   861‚Üí      // NOTE: vote_ended activity is sent by the client via activity:log.
   862‚Üí   862‚Üí
   863‚Üí   863‚Üí      console.log(`[vote:end] ${socket.id} -> ${roomId} winner=${winner}`)
   864‚Üí   864‚Üí    } catch (err) {
   865‚Üí   865‚Üí      console.error(`[vote:end] error from ${socket.id}:`, err)
   866‚Üí   866‚Üí      socket.emit('room:error', { event: 'vote:end', message: 'internal error' })
   867‚Üí   867‚Üí    }
   868‚Üí   868‚Üí  })
   869‚Üí   869‚Üí
   870‚Üí   870‚Üí  // ----- vote:cancel ------------------------------------------------------
   871‚Üí   871‚Üí  // Host cancels the active vote without announcing a winner. Just clears
   872‚Üí   872‚Üí  // state and broadcasts an inactive `vote:state`. Also logs a
   873‚Üí   873‚Üí  // "vote_cancelled" activity.
   874‚Üí   874‚Üí  socket.on('vote:cancel', (payload: unknown) => {
   875‚Üí   875‚Üí    try {
   876‚Üí   876‚Üí      const data = (payload ?? {}) as { roomId?: string }
   877‚Üí   877‚Üí      const roomId = typeof data.roomId === 'string' ? data.roomId.trim() : ''
   878‚Üí   878‚Üí      if (!roomId) return
   879‚Üí   879‚Üí
   880‚Üí   880‚Üí      const room = rooms.get(roomId)
   881‚Üí   881‚Üí      if (!room) return
   882‚Üí   882‚Üí
   883‚Üí   883‚Üí      if (room.host !== socket.id) {
   884‚Üí   884‚Üí        socket.emit('room:error', { event: 'vote:cancel', message: 'only host can cancel a vote' })
   885‚Üí   885‚Üí        return
   886‚Üí   886‚Üí      }
   887‚Üí   887‚Üí
   888‚Üí   888‚Üí      const wasItemId = room.vote?.itemId ?? ''
   889‚Üí   889‚Üí      room.vote = null
   890‚Üí   890‚Üí      io.to(roomId).emit('vote:state', voteStatePayload(room, nextSeq(room)))
   891‚Üí   891‚Üí
   892‚Üí   892‚Üí      // NOTE: vote_cancelled activity is sent by the client via activity:log.
   893‚Üí   893‚Üí
   894‚Üí   894‚Üí      console.log(`[vote:cancel] ${socket.id} -> ${roomId}`)
   895‚Üí   895‚Üí    } catch (err) {
   896‚Üí   896‚Üí      console.error(`[vote:cancel] error from ${socket.id}:`, err)
   897‚Üí   897‚Üí      socket.emit('room:error', { event: 'vote:cancel', message: 'internal error' })
   898‚Üí   898‚Üí    }
   899‚Üí   899‚Üí  })
   900‚Üí   900‚Üí
   901‚Üí   901‚Üí  // ----- disconnect ------------------------------------------------------
   902‚Üí   902‚Üí  // Remove the socket from every room it was in. For each affected room:
   903‚Üí   903‚Üí  //   - re-broadcast presence (with the possibly-new host),
   904‚Üí   904‚Üí  //   - push + broadcast a "left" activity,
   905‚Üí   905‚Üí  //   - if the user had voted on an active vote, scrub it and re-broadcast.
   906‚Üí   906‚Üí  // Finally clear the socket's identity.
   907‚Üí   907‚Üí  socket.on('disconnect', (reason: string) => {
   908‚Üí   908‚Üí    try {
   909‚Üí   909‚Üí      const joined = socketRooms.get(socket.id)
   910‚Üí   910‚Üí      if (joined) {
   911‚Üí   911‚Üí        for (const roomId of joined) {
   912‚Üí   912‚Üí          const room = rooms.get(roomId)
   913‚Üí   913‚Üí          // Capture the user BEFORE forgetRoom mutates the members Map.
   914‚Üí   914‚Üí          const user = room?.members.get(socket.id)
   915‚Üí   915‚Üí
   916‚Üí   916‚Üí          // Relay focus:clear to OTHER clients in this room so they remove
   917‚Üí   917‚Üí          // the disconnecting user's highlight. The socket is still valid
   918‚Üí   918‚Üí          // for one last `to()` emit during disconnect (it hasn't been
   919‚Üí   919‚Üí          // `leave()`'d yet), so we do this BEFORE forgetRoom/leave.
   920‚Üí   920‚Üí          try {
   921‚Üí   921‚Üí            socket.to(roomId).emit('focus:clear', { userId: socket.id })
   922‚Üí   922‚Üí          } catch {
   923‚Üí   923‚Üí            /* best-effort; never crash disconnect cleanup */
   924‚Üí   924‚Üí          }
   925‚Üí   925‚Üí
   926‚Üí   926‚Üí          forgetRoom(socket.id, roomId)
   927‚Üí   927‚Üí          socket.leave(roomId)
   928‚Üí   928‚Üí
   929‚Üí   929‚Üí          if (room) {
   930‚Üí   930‚Üí            // 1. Broadcast updated presence (includes new host id, if promoted).
   931‚Üí   931‚Üí            broadcastPresence(io, room, roomId)
   932‚Üí   932‚Üí
   933‚Üí   933‚Üí            // 2. Push + broadcast "left" activity.
   934‚Üí   934‚Üí            if (user) {
   935‚Üí   935‚Üí              broadcastActivity(io, room, roomId, {
   936‚Üí   936‚Üí                id: uuid(),
   937‚Üí   937‚Üí                userId: socket.id,
   938‚Üí   938‚Üí                userName: user.name,
   939‚Üí   939‚Üí                action: 'left',
   940‚Üí   940‚Üí                detail: user.name,
   941‚Üí   941‚Üí                ts: Date.now(),
   942‚Üí   942‚Üí              })
   943‚Üí   943‚Üí            }
   944‚Üí   944‚Üí
   945‚Üí   945‚Üí            // 3. If the disconnected user had voted on the active vote, drop
   946‚Üí   946‚Üí            //    their vote and re-broadcast the tally so the count is live.
   947‚Üí   947‚Üí            if (room.vote && room.vote.votes.has(socket.id)) {
   948‚Üí   948‚Üí              room.vote.votes.delete(socket.id)
   949‚Üí   949‚Üí              io.to(roomId).emit('vote:state', voteStatePayload(room, nextSeq(room)))
   950‚Üí   950‚Üí            }
   951‚Üí   951‚Üí          }
   952‚Üí   952‚Üí        }
   953‚Üí   953‚Üí        socketRooms.delete(socket.id)
